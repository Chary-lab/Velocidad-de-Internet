document.getElementById('themeToggle').addEventListener('click', toggleTheme); document.getElementById('langToggle').addEventListener('click', toggleLanguage); }); // ========== FUNCIONES DE TEST REAL ========== async function runTest() { const startBtn = document.getElementById('startBtn'); const statusEl = document.getElementById('testStatus'); const pingEl = document.getElementById('pingValue'); const downloadEl = document.getElementById('downloadValue'); const uploadEl = document.getElementById('uploadValue'); const gaugeCircle = document.querySelector('.gauge-circle'); // Deshabilitar botón durante la prueba startBtn.disabled = true; startBtn.textContent = t('starting'); gaugeCircle.classList.add('active'); // Resetear valores pingEl.textContent = '--'; downloadEl.textContent = '--'; uploadEl.textContent = '--'; statusEl.textContent = ''; let ping = 0, downloadSpeed = 0, uploadSpeed = 0; let testError = false; try { // Fase 1: Ping (promedio de 5 intentos) statusEl.textContent = t('pingPhase'); ping = await measurePing(5); pingEl.textContent = ping.toFixed(1); // Fase 2: Descarga statusEl.textContent = t('downloadPhase'); downloadSpeed = await measureDownload(); downloadEl.textContent = downloadSpeed.toFixed(2); // Fase 3: Subida statusEl.textContent = t('uploadPhase'); uploadSpeed = await measureUpload(); uploadEl.textContent = uploadSpeed.toFixed(2); } catch (error) { console.error('Error en el test:', error); statusEl.textContent = t('error'); testError = true; } gaugeCircle.classList.remove('active'); startBtn.disabled = false; startBtn.textContent = t('start'); if (!testError) { statusEl.textContent = t('complete'); saveResult(ping, downloadSpeed, uploadSpeed); updateCapacidadEnlace(downloadSpeed); updateGuiaCalidad(downloadSpeed); updateGamingPings(ping); } } // Medición de ping: solicitudes HEAD a un recurso pequeño async function measurePing(attempts = 5) { const url = 'https://speed.cloudflare.com/__down?bytes=0'; // recurso mínimo let total = 0; for (let i = 0; i < attempts; i++) { const start = performance.now(); await fetch(url, { method: 'HEAD', cache: 'no-store' }); const end = performance.now(); total += (end - start); } return total / attempts; // ping en ms } // Medición de descarga real async function measureDownload() { const startTime = performance.now(); const response = await fetch(`${CF_DOWNLOAD_URL}${DOWNLOAD_SIZE}`, { cache: 'no-store' }); const data = await response.arrayBuffer(); const endTime = performance.now(); const durationSeconds = (endTime - startTime) / 1000; const bitsLoaded = DOWNLOAD_SIZE * 8; const speedMbps = (bitsLoaded / durationSeconds) / 1_000_000; return speedMbps; } // Medición de subida real async function measureUpload() { // Generar datos aleatorios del tamaño exacto const randomData = new Uint8Array(UPLOAD_SIZE); for (let i = 0; i < UPLOAD_SIZE; i++) { randomData[i] = Math.floor(Math.random() * 256); } const blob = new Blob([randomData], { type: 'application/octet-stream' }); const startTime = performance.now(); const response = await fetch(CF_UPLOAD_URL, { method: 'POST', body: blob, cache: 'no-store', headers: { 'Content-Type': 'application/octet-stream' } }); const endTime = performance.now(); // Leer respuesta para asegurar que la subida terminó await response.text(); const durationSeconds = (endTime - startTime) / 1000; const bitsUploaded = UPLOAD_SIZE * 8; const speedMbps = (bitsUploaded / durationSeconds) / 1_000_000; return speedMbps; } // ========== FUNCIONES AUXILIARES (sin cambios mayores) ========== function t(key) { return translations[currentLang][key] || key; JSON.parse(stored) : []; displayResultsHistory(); } function displayResultsHistory() { const container = document.getElementById('historyList'); if (!container) return; if (resultsHistory.length === 0) { container.innerHTML = `
  • ${t('noHistory')}
    • `; return; } container.innerHTML = resultsHistory.map((r, i) => `
  • ${new Date(r.date).toLocaleString()} ⬇ ${r.download} Mbps / ⬆ ${r.upload} Mbps / 🕒 ${r.ping} ms
    • `).join(''); } function copyHistory(index) { const r = resultsHistory[index]; navigator.clipboard.writeText(`⬇ ${r.download} Mbps / ⬆ ${r.upload} Mbps / Ping ${r.ping} ms`); } // Gráfico (sin cambios relevantes) function renderChart() { const ctx = document.getElementById('speedChart').getContext('2d'); chart = new Chart(ctx, { type: 'line', data: { labels: [], datasets: [{ label: 'Descarga (Mbps)', data: [], borderColor: '#0d6efd', tension: 0.1 }, { label: 'Subida (Mbps)', data: [], borderColor: '#198754', tension: 0.1 }] } }); updateChart(); } function updateChart() { if (!chart) return; chart.data.labels = resultsHistory.map(r => new Date(r.date).toLocaleTimeString()); chart.data.datasets[0].data = resultsHistory.map(r => r.download); chart.data.datasets[1].data = resultsHistory.map(r => r.upload); chart.update(); } // Tema e idioma (sin cambios) function toggleTheme() { document.body.classList.toggle('dark-mode'); localStorage.setItem(DARK_MODE_KEY, document.body.classList.contains('dark-mode')); } function loadTheme() { if (localStorage.getItem(DARK_MODE_KEY) === 'true') { document.body.classList.add('dark-mode'); } } function toggleLanguage() { currentLang = currentLang === 'es' ?
